<script>
function loadCreateAccountModal() {
  const modalBody = document.querySelector('#createAccountModal .modal-body');
  fetch('modules/account_create.php')
    .then(response => response.json())
    .then(data => {
      if (data.form) {
        modalBody.innerHTML = data.form;
        attachFormSubmitHandler();
      } else if (data.success) {
        modalBody.innerHTML = `
          <div class="alert alert-success">
            Cuenta creada exitosamente.<br/>
            <strong>Account:</strong> ${data.accno}<br/>
            <strong>Password:</strong> ${data.password}
          </div>`;
        setTimeout(() => {
          // Close modal after successful account creation
          $('#createAccountModal').modal('hide');
        }, 2000);
      } else {
        modalBody.innerHTML = `<div class="alert alert-danger">${data.error}</div>`;
      }
    })
    .catch(error => {
      console.error('Error:', error);
      modalBody.innerHTML = '<div class="alert alert-danger">Error al cargar el formulario.</div>';
    });
}

function attachFormSubmitHandler() {
  const form = document.querySelector('#createAccountModal form');
  form.addEventListener('submit', function(e) {
    e.preventDefault();
    const formData = new FormData(form);

    fetch('modules/account_create.php', {
      method: 'POST',
      body: formData
    })
    .then(resp => resp.json())
    .then(data => {
      const modalBody = document.querySelector('#createAccountModal .modal-body');
      if (data.success) {
        modalBody.innerHTML = `
          <div class="alert alert-success">
            Cuenta creada exitosamente.<br/>
            <strong>Account:</strong> ${data.accno}<br/>
            <strong>Password:</strong> ${data.password}
          </div>`;
        setTimeout(() => {
          // Close modal after successful account creation
          $('#createAccountModal').modal('hide');
        }, 2000);
      } else {
        modalBody.innerHTML = `<div class="alert alert-danger">${data.error}</div>`;
      }
    })
    .catch(error => {
      console.error('Error:', error);
      document.querySelector('#createAccountModal .modal-body').innerHTML = '<div class="alert alert-danger">Error al crear la cuenta.</div>';
    });
  });
}
</script>
